003

009 * @version     1.0                 (current version number of program)
010 * @since       2016-03-26          (the version of the package this class was first added to)
011 */
012public class Stock {
013
014        List<Domino> pile;
015        
016        /**
017         * The constructor to be call by the user application. 
018         * It creates all the Dominos of the game and add them to the Stock.  
019         */                     
020        public Stock(){
021                
022                this.pile = new ArrayList<Domino>();
023                
024                //Creates a total of 28 Dominos. SUM(x)from 1 to 7 
025                for (int i=6; i>=0; i--){
026                        for (int j=i;j>=0;j--){
027                                this.addDomino(new Domino(i,j));
028                        }
029                }       
030        }
031        
032        /**
033         * Method to add the Domino d to the pile which is represented by a List of Dominos.
034         * @param d The Domino to be added.
035         */
036        private void addDomino(Domino d){
037                this.pile.add(d);
038        }
039        
040        /**
041         * Method to remove from the Stock the Domino that is in the List "pile" at the position n.
042         * @param n The index number of the Domino to be removed.
043         */
044        private void removeDomino(int n){
045                this.pile.remove(n);
046        }
047        
048        /**
049         * Method to obtain true or false if the Stock is empty or not.
050         * @return True if the pile is empty, otherwise false.
051         */
052        public boolean isEmpty(){
053                if(this.pile.isEmpty()){
054                        return true;
055                }
056                return false;
057        }
058        
059        /**
060         * Method to obtain a random Domino from the Stock.
061         * @return A random Domino from the Stock.
062         */
063        public Domino draw(){
064                Random rand = new Random();
065                int  n = rand.nextInt(this.pile.size());
066                
067                Domino d = this.pile.get(n);
068                this.removeDomino(n);
069                
070                return d;
071        }
072        
073        /**
074         * Method to obtain the String representation of the Stock.
075         * @return String representation of the Stock.
076         */     
077        public String toString(){
078                return this.pile.toString();
079        }
080}
