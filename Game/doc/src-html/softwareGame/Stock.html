014
015public class Stock<T> implements InterfaceStock<T> 
016{
017        @SuppressWarnings("rawtypes")
018        protected static Stock sto = null;
019        protected List<Domino<T>> pile = null;
020        
021        /**
022         * The constructor to be call by the Singleton getUniqueInstance method. 
023         * It creates all the Dominos of the game and add them to the Stock.  
024         */                     
025        private Stock(int t){
026                
027                this.pile = new ArrayList<Domino<T>>();
028                
029                //Creates a total of 28 Dominos. SUM(x)from 1 to 7 
030                for (int i=6; i>=0; i--)
031                {
032                        for (int j=i;j>=0;j--)
033                        {
034                                if(t==1)
035                                {
036                                        this.addDomino(new Domino(new Princess(i) ,new Princess(j)));
037                                        continue;
038                                }
039
040                                this.addDomino(new Domino(i,j));
041                        }
042                }
043                System.out.print(this.pile.size()+"\n");
044        }
045        
046        /**
047         * The constructor to be call by the user application to create the singleton stock.
048         * @return Singleton Stock instance.
049         */
050        @SuppressWarnings("rawtypes")
051        public static Stock getStock(int i){
052                
053                if(sto == null)
054                {
055                        sto = new Stock(i);
056                }       
057                return sto;             
058        }
059        
060        /**
061         * Method to add the Domino d to the pile which is represented by a List of Dominos.
062         * @param d The Domino to be added.
063         */
064        private void addDomino(Domino<T> d)
065        {
066                this.pile.add(d);
067        }
068        
069        /**
070         * Method to remove from the Stock the Domino that is in the List "pile" at the position n.
071         * @param n The index number of the Domino to be removed.
072         */
073        private void removeDomino(int n)
074        {
075                this.pile.remove(n);
076        }
077        
078        /**
079         * Method to obtain true or false if the Stock is empty or not.
080         * @return True if the pile is empty, otherwise false.
081         */
082        public boolean isEmpty()
083        {
084                if(this.pile.isEmpty())
085                {
086                        return true;
087                }
088                return false;
089        }
090        
091        /**
092         * Method to obtain a random Domino from the Stock.
093         * @return A random Domino from the Stock.
094         */
095        public Domino<T> draw()
096        {       
097                Random rand = new Random();
098                int  n = rand.nextInt(this.pile.size());
099                
100                Domino<T> d = this.pile.get(n);
101                this.removeDomino(n);
102                
103                return d;
104        }
105        
106        /**
107         * Method to obtain the String representation of the Stock.
108         * @return String representation of the Stock.
109         */     
110        public String toString()
111        {
112                return this.pile.toString();
113        }
114}
