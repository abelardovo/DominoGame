003import java.util.ArrayList;
004import java.util.List;
005

013 */
014
015public class Player {
016        
017        private String name;
018        private List<Domino> hand;
019        
020        /**
021         * The constructor to be called by the user application when creating the user's player.
022         * @param n Name of the user's player.
023         */
024        public Player(String n){
025                
026                this.name = n;
027                this.hand = new ArrayList<Domino>();
028                
029        }
030        
031        /**
032         * The constructor to be called by the user application when creating the PC.
033         */
034        public Player(){
035                
036                this.name = "Computer";
037                this.hand = new ArrayList<Domino>();
038        }
039        
040        /**
041         * Method to obtain name of the player. 
042         * @return The name of the player.
043         */
044        public String getName(){
045                return this.name;
046        }
047        
048        /**
049         * Method to obtain the size of the List "hand".
050         * @return The number of Dominos in the player's hand
051         */
052        public int getHandSize(){
053                return this.hand.size();
054        }
055
056        /**
057         * Method to obtain a certain Domino from the hand of the player. 
058         * @param n Index number that refers to the wanted Domino in the player's hand.
059         * @return The Domino in the player's hand, which index is input.
060         */
061        public Domino getDomino(int n){
062                return this.hand.get(n);
063        }
064        
065        /**
066         * Method to add a given Domino to the player's hand.
067         * @param d Domino to add to the the hand of the player.
068         */
069        public void addDomino(Domino d){
070                this.hand.add(d);
071        }       
072        
073        /**
074         * Method to remove a Domino from the player's hand, using an index number.
075         * @param n Index number that refers to the wanted Domino in the player's hand.
076         */
077        public void removeDomino(int n){
078                this.hand.remove(n);
079        }
080        
081        /**
082         * Method to remove a Domino from the player's hand, using the Domino the caller want to remove.
083         * @param d Domino that is to be removed from the player's hand.
084         */
085        public void removeDomino(Domino d){
086                
087                for(int n=0; n<this.hand.size(); n++){
088                        if(this.getDomino(n) == d){
089                                this.removeDomino(n);
090                        }               
091                }
092        }
093        
094        /**
095         * Method to determine if the last Domino chip has been played.
096         * @return true if the player's hand has no Dominos left, otherwise false.
097         */
098        public boolean noMoreDominos(){
099                
100                if(this.hand.isEmpty()){
101                        return true;
102                }
103                return false;
104        }
105        
106        /**
107         * Method that searches the player's hand, to determine if a given Domino is in the player's hand.
108         * @param left Left value of the Domino that is being searched.
109         * @param right Right value of the Domino being searched.
110         * @return true if the Domino given in the input is in the player's hand, otherwise false.
111         */
112        public boolean searchDomino(int left,int right){
113                
114                for(int n=0; n<this.hand.size(); n++){
115                        
116                        if( (right == this.getDomino(n).getLeftValue()) && (left == this.getDomino(n).getRightValue()) || 
117                                (right == this.getDomino(n).getRightValue()) && (left == this.getDomino(n).getLeftValue()) ){
118                                        
119                                return true;    
120                        }
121                }
122                return false;
123        }
124        
125        /**
126         * Method that searches the player's hand, to determine if a given Domino, that has double value, is in the player's hand.       
127         * @param i Value of the Domino being searched.
128         * @return If it is found, returns the index number "n" in the player's hand, of the Domino being searched, otherwise -1.
129         */
130        public int searchForDouble(int i){
131                
132                for(int n=0; n<this.hand.size(); n++){
133                        
134                        if(this.getDomino(n).isThereDouble()){
135        
136                                if(this.getDomino(n).getLeftValue() == i)
137                                        return n;
138                        }
139                        
140                }
141                return -1;
142        }
143
144        /**
145         * Method to obtain the String representation of a player's hand.
146         * @return String representation of the player's hand.
147         */
148        public String toString(){
149                String s = this.name + " hand: ";
150                for (int i=0; i<this.hand.size();i++){
151                        s = s + this.hand.get(i).toString() + " / ";
152                }
153                
154                s = s + "\n";
155
156                return s;
157        }
158        
159}
