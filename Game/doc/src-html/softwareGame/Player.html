001package softwareGame;
002
003import java.util.ArrayList;
004import java.util.Iterator;
005import java.util.List;
006
007/**
008 * CLASS PLAYER
009 * 
010 * @author      Abelardo Valino <abelardo.valino_ovalle@telecom-sudparis.eu>
011 * @author              Andres Gonzalez <andres.gonzalez_arria@telecom-sudparis.eu>
012 * @version     2.0                 (current version number of program)
013 * @since       2016-03-26          (the version of the package this class was first added to)
014 */
015
016public class Player<T> 
017{       
018        private String name;
019        private List<Domino<T>> hand;
020        
021        /**
022         * The constructor to be called by the user application when creating the user's player.
023         * @param n Name of the user's player.
024         */
025        public Player(String n)
026        {
027                this.name = n;
028                this.hand = new ArrayList<Domino<T>>();
029        }
030        
031        /**
032         * The constructor to be called by the user application when creating the PC.
033         */
034        public Player()
035        {       
036                this.name = "Computer";
037                this.hand = new ArrayList<Domino<T>>();
038        }
039        
040        /**
041         * Method to obtain name of the player. 
042         * @return The name of the player.
043         */
044        public String getName()
045        {
046                return this.name;
047        }
048        
049        /**
050         * Method to obtain the size of the List "hand".
051         * @return The number of Dominos in the player's hand
052         */
053        public int getHandSize()
054        {
055                return this.hand.size();
056        }
057
058        /**
059         * Method to obtain a certain Domino from the hand of the player. 
060         * @param n Index number that refers to the wanted Domino in the player's hand.
061         * @return The Domino in the player's hand, which index is input.
062         */
063        public Domino<T> getDomino(int n)
064        {
065                return this.hand.get(n);
066        }
067        
068        /**
069         * Method to add a given Domino to the player's hand.
070         * @param d Domino to add to the the hand of the player.
071         */
072        public void addDomino(Domino<T> d)
073        {
074                this.hand.add(d);
075        }       
076        
077        /**
078         * Method to remove a Domino from the player's hand, using an index number.
079         * @param n Index number that refers to the wanted Domino in the player's hand.
080         */
081        public void removeDomino(int n)
082        {
083                this.hand.remove(n);
084        }
085        
086        /**
087         * Method to remove a Domino from the player's hand, using the Domino the caller want to remove.
088         * @param d Domino that is to be removed from the player's hand.
089         */
090        public void removeDomino(Domino<T> d)
091        {       
092                for(int n=0; n<this.hand.size(); n++)
093                {
094                        if(this.getDomino(n) == d)
095                        {
096                                this.removeDomino(n);
097                        }               
098                }
099        }
100        
101        /**
102         * Method to determine if the last Domino chip has been played.
103         * @return true if the player's hand has no Dominos left, otherwise false.
104         */
105        public boolean noMoreDominos()
106        {       
107                if(this.hand.isEmpty())
108                {
109                        return true;
110                }
111                return false;
112        }
113        
114        /**
115         * Method that searches the player's hand, to determine if a given Domino is in the player's hand.
116         * @param left Left value of the Domino that is being searched.
117         * @param right Right value of the Domino being searched.
118         * @return true if the Domino given in the input is in the player's hand, otherwise false.
119         */
120        public boolean searchDomino(T left,T right)
121        {               
122                // Iterator for the player's hand.
123                Iterator<Domino<T>> handIterator = this.hand.iterator();
124                Domino<T> d; 
125                
126                while (handIterator.hasNext())
127                {       
128                        d = handIterator.next();
129                        if( (d.getLeftValue().equals(right)) && (d.getRightValue().equals(left)) || 
130                                (d.getRightValue().equals(right)) && (d.getLeftValue().equals(left)) )
131                        {                       
132                                        return true;            
133                        }                       
134                }
135                
136                return false;
137        }
138        
139        /**
140         * Method that searches the player's hand, to determine if a given Domino, that has double value, is in the player's hand.       
141         * @param indState Value of the Domino being searched.
142         * @return If it is found, returns the index number "n" in the player's hand, of the Domino being searched, otherwise -1.
143         */
144        public int searchForDouble(int indState)
145        {       
146                // Iterator for the player's hand.
147                Iterator<Domino<T>> handIterator = this.hand.iterator();
148                Domino<T> d; 
149                int n = 0;
150                
151                while (handIterator.hasNext())
152                {       
153                        d = handIterator.next();
154
155                        if(d.isThereDouble())
156                        {
157                                if( d.getLeftValue() instanceof Princess)
158                                {
159                                        if(d.getLeftValue().toString().equals(Princess.ObtainPrincess(indState)))
160                                                return n;
161                                }else if (d.getLeftValue() instanceof Integer)
162                                        if(d.getLeftValue().equals(indState))
163                                        return n;
164                        }
165                        n++;
166                }
167                
168                return -1;
169        }
170
171        /**
172         * Method to obtain the String representation of a player's hand.
173         * @return String representation of the player's hand.
174         */
175        public String toString()
176        {
177                String s = this.name + " hand: ";
178                
179                // Iterator for the player's hand.
180                Iterator<Domino<T>> handIterator = this.hand.iterator();
181                
182                while (handIterator.hasNext())
183                {
184                        s = s + handIterator.next().toString() + " / ";
185                }
186                
187                s = s + "\n";
188
189                return s;
190        }
191        
192}
