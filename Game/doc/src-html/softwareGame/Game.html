078   /**
079    * This method is called when an event is produced in the graphical interface.
080    * @param val The state of the game.
081    */
082   public void receivedMessage(int val)
083   {
084           System.out.println( "\nAn event has been produced. Event: "+ val +", for state: "+indState);
085           
086           switch (val)
087           {
088
089                   case GGame.DATA_NAME:
090                           this.initialize(this.gGame.getPlayerName());
091                           break;                  
092                   
093                   case GGame.PLAYRIGHT:
094                           playright = true;
095                   
096                   case GGame.PLAY: 
097                           switch (indState)
098                           {
099                                   //Treating the double cases
100                                   case 0: case 1: case 2: case 3: case 4: case 5: case 6:                 
101                                           this.treatDoubleAnswer((Domino) this.gGame.getDomino());
102                                           break;
103                                        
104                                   //Treating the cases blockedComputer and play
105                                   case 7: case 8:                         
106                                           this.treatAnswer((Domino) this.gGame.getDomino());
107                                           break; 
108                                   
109                                   //Treating the case noDoubleFirstDomino
110                                   case 12:        
111                                           this.table.setValue((Domino) this.gGame.getDomino());
112                                           this.treatAnswer((Domino) this.gGame.getDomino());
113                                           break;
114                           }
115                           break;
116                           
117                   case GGame.JUMP:                
118                   switch (indState)
119                   {
120                           //Treating the double cases             
121                           case 0: case 1: case 2: case 3: case 4: case 5: case 6:                 
122                                   this.treatJumpAnswer();                 
123                                   break;
124                
125                           //Treating the case play when the stock is empty and the player wants to jump.
126                           case 8:                 
127                                   if(!this.player1.searchDomino(this.table.getLeftValue(), this.table.getRightValue()))
128                                   {
129                                           this.indState = 9;
130                                   }
131                                   
132                                   this.gGame.setEnabledJump(false);
133                                   this.gGame.setEnabledPlayPC(true);
134                                   break;
135                           }   
136                   break;
137                   
138                   case GGame.DRAW:                
139                           switch (indState)
140                           {
141                                   //Treating the case play when the player wants to DRAW from the Stock
142                                   case 8: 
143                                           this.playerDraw();
144                                           break;
145                           }
146                   break;                  
147
148                   case GGame.VALIDPCPLAY:
149                           this.computerPlay();
150                           break;
151           }
152           
153           //Prints by console the State of the board, the PC's hand and the Dominos that are on the table.
154           System.out.print(this.table.printState());      
155           System.out.print(this.pc.toString());
156           System.out.print("TABLE: "+this.table.toString()+"\n");         
157   }

195   /**
196    * Take the selected domino, verify if it is the right one.
197    * If it is ok, call treatAnswer method otherwise send a message.
198    * @param d The selected domino.
199    */
200   public void treatDoubleAnswer(Domino<T> d)
201   {
202           
203           //If the selected Domino is not the double needed.
204           if(this.GameType == 0){
205                   if (!(d.getLeftValue().equals(this.indState)) || !(d.getRightValue().equals(this.indState))){
206                           this.gGame.setMessage("This it is not the double "+this.indState);
207                           return;                         
208                   }
209           }
210           else if (this.GameType == 1)
211                   if (!(d.getLeftValue().toString().equals(Princess.ObtainPrincess(this.indState))) || !(d.getRightValue().toString().equals(Princess.ObtainPrincess((this.indState))))){
212                           this.gGame.setMessage("This it is not the double "+Princess.ObtainPrincess(this.indState));
213                           return;                         
214                   }
215           
216           this.indState = 8;
217           
218           //Set the right and left value of the Table.
219           this.table.setValue(d);
220           this.treatAnswer(d);
221           
222   }

225   /**
226    * When a player plays, we verify that the selected domino d may be
227    * put on the table. If it is the case, we remove d from the hand of the player
228    * and from the graphical hand. Then put d on the board and on the graphic board.
229    * @param d The domino selected by the player.
230    */
231    public void treatAnswer(Domino<T> d)
232    {
233        //Verify if the Domino cannot be played.
234        if(!this.table.canPlay(d))
235        {
236                this.gGame.setMessage("Can not play this Domino! ");
237                        return;
238                 }
239        
240        //Otherwise, we put the Domino on the table and remove it from the player's hand.
241        System.out.println("The player played: "+d.toString());
242        
243        if(playright && (this.table.canPlayRight(d)))
244        {
245                playright = false;
246                try {
247                                this.gGame.putDominoOnRightTable(d);
248                        this.table.playRight(d);
249                        } catch (BadMatchException e) {
250                                try {
251                                        this.gGame.putDominoOnLeftTable(d);
252                                } catch (BadMatchException e1) {
253                                        e1.printStackTrace();
254                                }
255                                this.table.play(d);
256                        };
257
258        }else{
259                try {
260                                this.gGame.putDominoOnLeftTable(d);
261                        this.table.play(d);
262
263                        } catch (BadMatchException e) {
264                                try {
265                                        this.gGame.putDominoOnRightTable(d);
266                                } catch (BadMatchException e1) {
267                                        e1.printStackTrace();
268                                }
269                        this.table.playRight(d);
270
271                        };
272
273        }
274        
275                //this.gGame.putDominoOnTable(d);
276                this.player1.removeDomino(d);
277                this.gGame.removeDominoFromHand(d);
278                
279                //If the state of the game was NoDoubleFirstDomino or BlockedComputer
280                //we change the state of the game to Play.
281                if((this.indState == 12)||(this.indState == 7))
282                {
283                   this.indState = 8;
284                }
285                
286                //We turn off the Jump, Draw buttons and disable the Graphical Hand
287                this.gGame.setEnabledJump(false);
288                this.gGame.setEnabledDraw(false);
289                this.gGame.setHandEnable(false);
290                
291                //Verifying if the player wins the game.
292                //It depends on if he has more Dominos in his hand or not.
293                if(this.player1.noMoreDominos())
294                {       
295                        this.indState = 10;
296                        
297                        this.gGame.setEnabledPlayPC(false);
298                        this.gGame.setMessage("You win!!!!");
299
300                }else{
301                        //Computer Play
302                        this.gGame.setEnabledPlayPC(true);
303                        this.gGame.setMessage("Click on Play PC, for the PC to play.");
304                }
305                
306    }

308        /**
309        * The player draw. If the stock is empty, the player has to choose a 
310        * Domino from his hand or Jump. 
311        * otherwise the player can draw a Domino from the Stock until he can play.
312        *
313        */
314        public void playerDraw()
315        { 
316                //If the Stock is empty, the player has to choose a Domino from his hand or jump.
317                if (this.stock.isEmpty())
318                {          
319                        this.gGame.setMessage("The Stock is empty. Please choose a domino or jump. ");
320                           
321                        this.gGame.setEnabledJump(true);
322                        this.gGame.setHandEnable(true);
323                        this.gGame.setEnabledDraw(false);
324                        this.gGame.setEnabledPlayPC(false);
325                        return;
326                }
327                
328                //Otherwise, the player can take a Domino from the Stock
329                Domino d = this.stock.draw();
330                this.player1.addDomino(d);
331                this.gGame.addDominoInHand(d);
332                
333        }

335        /**
336         * The computer plays in relationship with the state of the game.
337         * If state = n (with n =1,2,3,4,5,6) we look for a double n in the computer's hand.
338         * If yes, the computer plays else the player is asked to play the double domino (n-1).
339         * If n=0 we look for a double 0 in the computer's hand.
340         * If yes, the computer plays, otherwise the player is asked to play any other domino.
341         * If n=8 or 9  normal game managing the stock and the empty stock.
342         * If n=11 blocked game.
343         */
344        public void computerPlay( )
345        {
346                int i;
347                System.out.println("Computer tries to play, with state: "+indState);
348                Domino<T> d=null;
349
350                switch (indState)
351                {
352                 //we look for a double n in the computer's hand
353                 //If yes, the computer plays else the player is asked to play the double domino (n-1)
354                case 1: case 2: case 3: case 4: case 5: case 6: 
355                         
356                        i = this.pc.searchForDouble(indState);
357                        //If the computer has the Double requested, he plays it.
358                        if(i != -1)
359                        {
360                                this.table.initialPlay(this.pc.getDomino(i));
361                                this.gGame.putDominoOnTable(this.pc.getDomino(i));
362                                this.pc.removeDomino(i);
363                                
364                                this.indState = 8;
365                                
366                                this.gGame.setEnabledJump(false); 
367                                this.gGame.setEnabledPlayPC(false);
368                                this.gGame.setHandEnable(true);
369                                this.gGame.setEnabledDraw(true);
370                                 
371                                //Otherwise, he changes the state of the game to look for the next double.      
372                        }else{
373                                
374                                this.indState--;
375        
376                                if (this.GameType == 0)
377                                        this.gGame.setMessage( "The PC does not have the double. Please click on double "+this.indState+" or jump");
378                                else if (this.GameType == 1)
379                                        this.gGame.setMessage( "The PC does not have the double. Please click on double "+Princess.ObtainPrincess(this.indState)+" or jump");                                   
380                                
381                                this.gGame.setEnabledPlayPC(false);
382                                this.gGame.setEnabledDraw(false);
383                                this.gGame.setHandEnable(true);
384                                this.gGame.setEnabledJump(true);
385                                }
386                                
387                                break;
388 
389                 //If n=0 we look for a double 0 in the computer's hand.
390                 //If true, the computer plays, otherwise the player is asked to play any other domino.
391                 case 0:
392                         i = this.pc.searchForDouble(indState); 
393                 
394                         //If the computer has the Double 0, he plays it.
395                         if(i != -1)
396                         {
397                                this.table.initialPlay(this.pc.getDomino(i));
398                                this.gGame.putDominoOnTable(this.pc.getDomino(i));
399                                this.pc.removeDomino(i);
400                                
401                                this.indState = 8;
402        
403                                this.gGame.setEnabledJump(false); 
404                                this.gGame.setEnabledPlayPC(false);
405                                this.gGame.setHandEnable(true);
406                                this.gGame.setEnabledDraw(true);                                 
407                        //Otherwise, the computer changes the state of the game to NoDoubleFirstDomino.
408                        }else{
409                                this.indState = 12;
410                                
411                                this.gGame.setMessage( "Please click on any domino.");
412                                
413                                this.gGame.setEnabledPlayPC(false);
414                                this.gGame.setEnabledJump(false);
415                                this.gGame.setEnabledDraw(false);
416                                this.gGame.setHandEnable(true); 
417                        }
418                
419                         break;
420        
421                //Case if the Computer is blocked.
422                case 7:
423                        this.gGame.setMessage("The Computer is blocked! Play again.");
424                        this.gGame.setEnabledPlayPC(false);
425                        this.gGame.setHandEnable(true);
426                        this.gGame.setEnabledJump(false); 
427                        this.gGame.setEnabledDraw(true);                         
428                        break;
429                
430                 //If n=8 or 9  normal game managing the stock and the empty stock.
431                 case 8:case 9:
432                         i=0;
433                
434                         //Verifying which Domino can play the computer.
435                         while(!this.table.canPlay(this.pc.getDomino(i)))
436                         {
437                                 i++;
438                        
439                                 //If the verification reaches the last Domino in the Computer's hand
440                                 if(i>= this.pc.getHandSize())
441                                 {
442                                         //We check if the Stock is empty. If true, it means that the PC cannot Draw more Dominos.
443                                         if(this.stock.isEmpty())
444                                         {
445                                                 //If the state of the game was BlockedPlayer and the PC cannot play any Domino, it's a DRAW!
446                                                 if(this.indState == 9)
447                                                 {
448                                                         this.indState = 11;
449                                                         this.gGame.setMessage("The Stock is empty, the PC cannot play. It's a DRAW!");                                                 
450                                                         break;
451                                                 }
452                                        
453                                                 //Otherwise, the state of the game changes to BlockedComputer
454                                                 this.indState = 7;
455                                                 this.gGame.setMessage("The Stock is empty and the PC cannot play. Select a domino!");
456                                                 break;
457                                         }
458                                
459                                         //Otherwise, the Computer can draw a Domino from the Stock.
460                                         this.pc.addDomino( this.stock.draw() );
461                                 }
462                         }
463                
464                        //If the state of the game changes to BlockedGame, the game is over so we disable all the
465                        //buttons and the player's hand.
466                        if (this.indState == 11)
467                        {
468                                this.gGame.setHandEnable(false);
469                                this.gGame.setEnabledDraw(false);
470                                this.gGame.setEnabledPlayPC(false);
471                                this.gGame.setEnabledJump(false); 
472                                break;
473                        }
474                
475                        //If the state of the game changes to BlockedComputer, 
476                        //we disable all the buttons and enable the playe's hand.
477                        if (this.indState == 7)
478                        {
479                                this.gGame.setEnabledDraw(false);
480                                this.gGame.setEnabledPlayPC(false);
481                                this.gGame.setEnabledJump(false);
482                                this.gGame.setHandEnable(true);
483                                break;
484                        }               
485                
486                        //If the state of the game remains the same, the Computer plays the next available Domino. 
487                        d = this.pc.getDomino(i);
488                        this.table.play(d);
489                        System.out.println("The computer has played: "+ d.toString() );
490                        this.gGame.putDominoOnTable(d);
491                        this.pc.removeDomino(i);
492                        
493                        //If the Computer doesn't have any Domino left, it means that it wins.
494                        if(this.pc.noMoreDominos())
495                        {
496                                this.indState = 10;
497                                
498                                this.gGame.setEnabledPlayPC(false);
499                                this.gGame.setEnabledJump(false); 
500                                this.gGame.setHandEnable(false);
501                                this.gGame.setEnabledDraw(false);
502                                
503                                this.gGame.setMessage("PC wins!!!!");
504        
505                        //Otherwise, the game continues.
506                        }else{
507                                this.gGame.setEnabledPlayPC(false);
508                                this.gGame.setHandEnable(true);
509                        
510                                //If the Stock is empty, we disable the Draw button.
511                                if(this.stock.isEmpty())
512                                {
513        
514                                        this.indState = 8;
515        
516                                        this.gGame.setEnabledJump(true); 
517                                        this.gGame.setEnabledDraw(false);
518                                        this.gGame.setMessage("Stock empty. "+this.player1.getName()+" choose a domino or JUMP.");
519        
520                                //Otherwise, we activate the Draw button and the Player's hand.
521                                }else{
522                                        this.gGame.setEnabledJump(false); 
523                                        this.gGame.setEnabledDraw(true);
524                                        this.gGame.setMessage("The PC played "+d.toString()+".\n"+this.player1.getName()+" choose a domino or DRAW from the Stock.");
525
526                                }                               
527                        }
528                
529                break;
530                 
531        default: 
532                System.out.println("state no valid");                   
533         }              
534}

544                        case 0: case 1: case 2: case 3: case 4: case 5: case 6:
545                   
546                                 //If the Player does not have the double requested, it's the PC's turn.

549                                         this.gGame.setHandEnable(false);

556                                 }else{
557                                         if(this.GameType == 0)

561                                 }
562                        
563                        break;
564         }
