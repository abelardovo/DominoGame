008 * @version     1.0                 (current version number of program)
009 * @since       2016-05-08          (the version of the package this class was first added to)
010 */
011public class Princess {
012        
013        private String name;
014
015        /**
016         * The constructor to be call by the Stock constructor to create all the Dominos<Princesses>.
017         * @param i the value of the domino.
018         */             
019        public Princess(int i){
020                
021                this.name = ObtainPrincess(i);
022        }
023        
024        /**
025         * Method to obtain the name of a princess.
026         * @param i the value of the domino.
027         */     
028        public static String ObtainPrincess(int i){
029                
030                switch (i)
031                {
032                case 0:
033                        return "Jasmine";
034                case 1:
035                        return "Aurora";
036                case 2:
037                        return "Cinderella";
038                case 3:
039                        return "Ariel";
040                case 4:
041                        return "Belle";
042                case 5:
043                        return "SnowWhite";
044                case 6:
045                        return "Kairi";
046                }
047                return "ERROR";
048        }
049        
050        /**
051         * Method to compare two princesses.
052         * @param p a princess.
053         */
054        @Override
055        public boolean equals(Object p){
056                
057                Princess paux = (Princess) p;
058                return this.name == paux.name;
059        }
060        
061        /**
062         * Method to obtain the String representation of a Princess.
063         * @return String representation of the Princess.
064         */
065        public String toString(){
066                
067                return this.name;
068        }
069}
