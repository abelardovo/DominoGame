001package softwareGame;
002
003import java.util.ArrayList;
004import java.util.List;
005
006/**
007 * CLASS TABLE

009 * @author      Abelardo Valino <abelardo.valino_ovalle@telecom-sudparis.eu>
010 * @author              Andres Gonzalez <andres.gonzalez_arria@telecom-sudparis.eu>
011 * @version     1.0                 (current version number of program)
012 * @since       2016-03-26          (the version of the package this class was first added to)
013 */
014
015public class Table {
016        
017        List<Domino> board;
018        int right;
019        int left;
020
021        /**
022         * The constructor to be called by the user application when creating the table in which to play.
023         */
024        public Table(){
025                this.board = new ArrayList<Domino>();
026                this.right = -1;
027                this.left = -1;
028        }
029        
030        /**
031         * Method to obtain the right value of the table, which is one of the two possible values a Domino must have in order to be played.
032         * @return Right value of the table.
033         */
034        public int getRightValue(){
035                return this.right;
036        }
037        
038        /**
039         * Method to obtain the left value of the table, which is one of the two possible values a Domino must have in order to be played.
040         * @return Left value of the table.
041         */
042        public int getLeftValue(){
043                return this.left;
044        }
045        
046        /**
047         * Method to set initial values to the right and left variables of the table.
048         * @param d Domino that is going to be added to the table, which values are extracted so it will be an acceptable Domino.
049         */
050        public void setValue(Domino d){
051                this.right = d.getRightValue();
052                this.left = d.getLeftValue();
053        }
054        
055        /**
056         * Method that checks if a given Domino can be played in the Table. One of its values match with the left or the right value of the table.
057         * @param d Domino that is being verified to confirm it can be played.
058         * @return true if the Domino can be played, otherwise false.
059         */
060        public boolean canPlay(Domino d){
061                
062                if( (this.right == d.getLeftValue()) || (this.right == d.getRightValue()) ||
063                        (this.left == d.getLeftValue())  || (this.left == d.getRightValue())){
064                        return true;
065                }
066                
067                return false;   
068        }
069        
070        /**
071         * Method to set the initial Domino in the table.
072         * @param d Domino to be setted in the table.
073         */
074        public void initialPlay(Domino d){
075                
076                this.board.add(d);
077                this.left = d.getLeftValue();
078                this.right = d.getRightValue();
079        
080        }
081        
082        /**
083         * Method that adds a given Domino to the table. By default, the given Domino will be added in the left side of the table.
084         * @param d Domino to be added in the table.
085         */
086        public void play(Domino d){
087                // Checking if values of Domino match the left value of the table. 
088                if(this.left == d.getRightValue()){
089                        this.board.add(0,d);
090                        this.left = d.getLeftValue();
091                        return;
092                }
093                
094                if(this.left == d.getLeftValue()){
095                        this.board.add(0,d);
096                        this.left = d.getRightValue();
097                        return;
098                }
099
100                // Checking if values of Domino match the right value of the table. 
101                if(this.right == d.getRightValue()){
102                        this.board.add(d);
103                        this.right = d.getLeftValue();
104                        return;
105                }
106                
107                if(this.right == d.getLeftValue()){
108                        this.board.add(d);
109                        this.right = d.getRightValue();
110                        return;
111                }               
112                
113                
114        }
115        
116        /**
117         * Method to obtain the String representation of table's state (left and right values).
118         * @return String representation of the table's state.
119         */
120        public String printState(){
121                return "STATE TABLE: "+this.getLeftValue()+" : "+this.getRightValue()+"\n";
122        }
123        
124        /**
125         * Method to obtain the String representation of Dominos in the table.
126         * @return String representation of the Dominos in the table.
127         */
128        public String toString(){
129                return this.board.toString();
130        }
131                
132}
