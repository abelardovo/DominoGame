001package softwareGame;
002import graphicInterface.InterfaceDomino;
003
004/**
005 * CLASS DOMINO
006 * 
007 * @author      Abelardo Valino <abelardo.valino_ovalle@telecom-sudparis.eu>
008 * @author              Andres Gonzalez <andres.gonzalez_arria@telecom-sudparis.eu>
009 * @version     1.0                 (current version number of program)
010 * @since       2016-03-26          (the version of the package this class was first added to)
011 */
012public class Domino implements InterfaceDomino{
013        
014        private int right;
015        private int left;
016        
017        /**
018         * The constructor to be call by the Stock constructor to create all the Dominos.
019         * @param r The right value of the Domino.
020         * @param l The left value of the Domino. 
021         */             
022        public Domino(int r, int l){
023                this.right = r;
024                this.left = l;
025        }
026        
027        /**
028         * Method to obtain the right value of a Domino.
029         * @return the right value of the Domino.
030         */             
031        public int getRightValue(){
032                return this.right;
033        }
034        
035        /**
036         * Method to obtain the left value of a Domino.
037         * @return Returns the left value of the Domino.
038         */     
039        public int getLeftValue(){
040                return this.left;
041        }       
042        
043        /**
044         * Method to obtain true or false if the Domino is a double or not.
045         * @return true if the left and right value of a Domino are equal, otherwise false.
046         */             
047        public boolean isThereDouble(){
048                if(this.getLeftValue() == this.getRightValue())
049                        return true;
050                return false;
051        }
052        
053        /**
054         * Method to obtain the String representation of a Domino.
055         * @return String representation of the Domino.
056         */
057        public String toString(){
058                return Integer.toString(this.left)+":"+Integer.toString(this.right);
059        }
060}
